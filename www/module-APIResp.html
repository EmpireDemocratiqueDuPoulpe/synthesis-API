Example
const response = new APIResp(200).setMessage("Logged in!").setData({ user: {firstName: "Hammong", lastName: "Gus"} });
const json = response.toJSON();
console.log(json); // ->
// {
//   code: 200,
//   message: "Logged in!",
//   data: {
//     firstName: "Hammong",
//     lastName: "Gus"
//   }
// }
